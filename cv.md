function descendingOrder(n){
    let array = []
    let nCopy = n
    while (nCopy >= 0) {
        let digit = nCopy % 10
        array.push(digit)
        nCopy = Math.floor(nCopy / 10)
        if (nCopy === 0) {
            break
        }
    }
    array.sort( (a,b) => (a < b)? 1 : -1)
    return Number(array.join(''));
}
